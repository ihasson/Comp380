012import java.util.ArrayList;

015/*
016 * This class holds all the courses associated with a given major.
017 * When the Major class is instantiated, it will contiain an ArrayList
018 * of Courses (for this - using the Sections objects).
019 * <p>
020 * The methods for the Major class are:
021 * addMajorReq
022 * displayMajorReqs
023 * 
024 */
025public class Major {
026
027        private int majorID;//holds the id for a Major
028        private String majorName;//holds the name of Major
029        
030        ArrayList<Course> majorReqs = //holds the required Courses
031                        new ArrayList<Course>();//for a Major
032        
033        /* 
034         * Major constructor
035         * 
036         * @param id MajorID
037         * @param major Major name
038         * @param courses ArrayList of the major requirement courses
039         * 
040         */
041        Major(int id, String major, ArrayList<Course> courses){
042                
043                majorID = id;
044                majorName = major;
045                majorReqs = courses;
046        }
047        
048        /*
049         * Adds a Course object to the requirements list of courses for a Major
050         * 
051         * @param course the Course object
052         * @return void
053         */
054        private void addMajorReq(Course course){
055                
056                majorReqs.add(course);
057                
058        }
059        
060        /*
061         * Displays the ArrayList of Major required courses.
062         * 
063         * @param none
064         * @return void
065         */
066        public void displayMajorReqs(){
067                
068                for(int index = 0; index < majorReqs.size(); index++){
069                        
070                        if(!majorReqs.isEmpty()){
071                                System.out.println(majorReqs.get(index));
072                        }
073                        else{
074                                System.out.println("There are no Courses for this Major");
075                        }
076                        
077                }
078        }
079        
080        public String getName(){
081                return majorName;
082        }
083        
084        public ArrayList<Course> getMajorReqs(){
085                return majorReqs;
086        }
087}
