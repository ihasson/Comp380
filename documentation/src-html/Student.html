023import java.util.LinkedList;
024
025public class Student {
026        
027//variables:
028        private String studentId;
029        private String firstName;
030        private String lastName;
031        private String studentMonthOfAdmit;
032        private Major studentMajor;
033        private Schedule currentSchedule;
034        private Planner studentPlaner;
035        private LinkedList<Course> studentTransscript;
036        private char middleInitial;
037        private int studentYearOfAdmit;
038        
039//Constructors:
040        
041        public Student(){
042                this.studentId = "";
043                this.firstName = "";
044                this.lastName = "";
045                this.middleInitial = ' ';
046                this.studentMonthOfAdmit = "";
047                this.studentYearOfAdmit = 0;
048                this.studentTransscript = new LinkedList<Course>();
049        }//end
050        
051        public Student(String sId, String sFirstName, String sLastName, char sInit){
052                this.studentId = sId;
053                this.firstName = sFirstName;
054                this.lastName = sLastName;
055                this.middleInitial = sInit;
056                this.studentTransscript = new LinkedList<Course>();
057                this.studentMonthOfAdmit = "not set";
058                this.studentYearOfAdmit = 0;
059        }//end
060        
061//public Methods:
062        
063        /*
064         * adds a course to the list.
065         */
066        public void updatTransscript(Course input){
067                this.studentTransscript.add(input);
068        }//end 
069        
070        /*
071         * returns a course from the list at
072         * the given index i.
073         */
074        public Course getCorsefromTransscript(int i){
075                return this.studentTransscript.get(i);
076        }//end
077        
078        /*
079         * returns a boolean if the given course 
080         * has been taken.
081         */
082        public boolean courseTaken(Course input){
083                if(this.studentTransscript.contains(input)){
084                        return true;
085                }//end if
086                return false;
087        }//end 
088        
089        /*
090         * returns the list as an array of courses
091         */
092        public Course[] courseLodeout(){
093                return (Course[]) this.studentTransscript.toArray();
094        }//end
095        
096//Get and Set methods:
097
098        public String getStudentId() {
099                return studentId;
100        }
101        public void setStudentId(String studentId) {
102                this.studentId = studentId;
103        }
104        public String getFirstName() {
105                return firstName;
106        }
107        public void setFirstName(String firstName) {
108                this.firstName = firstName;
109        }
110        public String getLastName() {
111                return lastName;
112        }
113        public void setLastName(String lastName) {
114                this.lastName = lastName;
115        }
116        public String getStudentMonthOfAdmit() {
117                return studentMonthOfAdmit;
118        }
119        public void setStudentMonthOfAdmit(String studentMonthOfAdmit) {
120                this.studentMonthOfAdmit = studentMonthOfAdmit;
121        }
122        public Major getStudentMajor() {
123                return studentMajor;
124        }
125        public void setStudentMajor(Major studentMajor) {
126                this.studentMajor = studentMajor;
127        }
128        public Schedule getCurrentSchedule() {
129                return currentSchedule;
130        }
131        public void setCurrentSchedule(Schedule currentSchedule) {
132                this.currentSchedule = currentSchedule;
133        }
134        public Planner getStudentPlaner() {
135                return studentPlaner;
136        }
137        public void setStudentPlaner(Planner studentPlaner) {
138                this.studentPlaner = studentPlaner;
139        }
140        public char getMiddleInitial() {
141                return middleInitial;
142        }
143        public void setMiddleInitial(char middleInitial) {
144                this.middleInitial = middleInitial;
145        }
146        public int getStudentYearOfAdmit() {
147                return studentYearOfAdmit;
148        }
149        public void setStudentYearOfAdmit(int studentYearOfAdmit) {
150                this.studentYearOfAdmit = studentYearOfAdmit;
151        }
152        
153//Override Methods: 
154
155        public String toString(){
156                return "Student: " + this.studentId + " " + this.firstName + " " + this.lastName + " " + this.middleInitial;
157        }//end
158}//end Student class
