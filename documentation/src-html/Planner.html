077    // ensures PlannerUnits are stored chronologically so that the earliest
078    // information available is quickly accessed
079    public void orderChornologically() {
080        boolean sorted = false;
081        
082        for(int pass = 1; (pass<allUnits.size()) && !sorted; ++pass) {
083            sorted = true; // assume sorted
084            for(int index = 0; index<allUnits.size()-pass; ++index) {
085                int nextIndex = index+1;
086                if(allUnits.get(index).sortValue() >
087                        allUnits.get(nextIndex).sortValue()) {
088                    // if former is greater than latter, exchange items
089                    PlannerUnit temp = allUnits.get(index);
090                    
091                    allUnits.get(index).
092                            setSemester(allUnits.get(nextIndex).getSemester());
093                    allUnits.get(index).
094                            setYear(allUnits.get(nextIndex).getYear());
095                    allUnits.get(index).
096                            setMaxUnits(allUnits.get(nextIndex).getMaxUnits());
097                    allUnits.get(index).
098                            setDifficultyRating(allUnits.get(nextIndex).
099                                    getDifficultyRating());
100                    
101                    allUnits.get(nextIndex).setSemester(temp.getSemester());
102                    allUnits.get(nextIndex).setYear(temp.getYear());
103                    allUnits.get(nextIndex).setMaxUnits(temp.getMaxUnits());
104                    allUnits.get(nextIndex).
105                            setDifficultyRating(temp.getDifficultyRating());
106
107                    sorted = false; // signal exchange
108                } // end if
109            } // end for
110        } // end for
111    } // end orderChronologically

113    // calculates the difficulty of a course
114    public double calculateDifficultyRating(String cID) throws IOException {
115        String ID;
116        String courseID;
117        String semester;
118        String year;
119        String hoursPerWeek;
120        String grade;
121        
122        int totalHours = 0;
123        int countHours = 0;
124        double avgHours = 0;
125        
126        double totalGrade = 0;
127        int countGrade = 0;
128        double avgGrade = 0;
129        
130        // reads the hoursSurveyData text file to record total hours estimated
131        // as well as the number of data points
132        // ultimately used to calculate the average, using total/count
133        File hoursFile = new File("C:\\Users\\Andrea Lee\\Documents\\NetBeansProjects\\Planner\\hoursSurveyData.txt");
134        Scanner hoursInput = new Scanner(hoursFile);
135
136        while (hoursInput.hasNextLine()) {
137            ID = hoursInput.next();
138            courseID = hoursInput.next();
139            semester = hoursInput.next();
140            year = hoursInput.next();
141            hoursPerWeek = hoursInput.next();
142            
143            if (cID.equals(courseID))
144            {
145                countHours++;
146                totalHours = totalHours + Integer.parseInt(hoursPerWeek);
147            }
148        }
149        
150        hoursInput.close();
151        
152        // reads the gradeDistribution text file to record total grade points
153        // with an A being 4.00, a B being 3.00, and so forth
154        // as well as the number of data points
155        // ultimately used to calculate the average, using total/count
156        File gradeFile = new File("C:\\Users\\Andrea Lee\\Documents\\NetBeansProjects\\Planner\\gradeDistribution.txt");
157        Scanner gradeInput = new Scanner(gradeFile);
158
159        while (gradeInput.hasNextLine()) {
160            ID = gradeInput.next();
161            courseID = gradeInput.next();
162            semester = gradeInput.next();
163            year = gradeInput.next();
164            grade = gradeInput.next();
165            
166            if (cID.equals(courseID))
167            {
168                countGrade++;
169                
170                if(grade.equals("A+")) {
171                    totalGrade = totalGrade + 4.33;
172                }
173                else if(grade.equals("A")) {
174                    totalGrade = totalGrade + 4.00;
175                }
176                else if(grade.equals("A-")) {
177                    totalGrade = totalGrade + 3.67;
178                }
179                else if(grade.equals("B+")) {
180                    totalGrade = totalGrade + 3.33;
181                }
182                else if(grade.equals("B")) {
183                    totalGrade = totalGrade + 3.00;
184                }
185                else if(grade.equals("B-")) {
186                    totalGrade = totalGrade + 2.67;
187                }
188                else if(grade.equals("C+")) {
189                    totalGrade = totalGrade + 2.33;
190                }
191                else if(grade.equals("C")) {
192                    totalGrade = totalGrade + 2.00;
193                }
194                else if(grade.equals("C-")) {
195                    totalGrade = totalGrade + 1.67;
196                }
197                else if(grade.equals("D+")) {
198                    totalGrade = totalGrade + 1.33;
199                }
200                else if(grade.equals("D")) {
201                    totalGrade = totalGrade + 1.00;
202                }
203                else if(grade.equals("D-")) {
204                    totalGrade = totalGrade + 0.67;
205                }
206                else {
207                    totalGrade = totalGrade + 0.00;
208                }
209            }
210        }
211        
212        gradeInput.close();
213        
214        // the average is calculated; assumptions are made if no data was
215        // available, with average hours defaulting to the course's unit
216        // count, and average grade defaulting to a C, or 2.00
217        if (totalHours > 0) {
218            avgHours = totalHours/countHours;
219        }
220        else {
221            //avgHours = course.getUnit(cID);
222        }
223        
224        if (totalGrade > 0) {
225            avgGrade = totalGrade/countGrade;
226        }
227        else {
228            avgGrade = 2.00;
229        }
230        
231        return avgHours/avgGrade;
232    } // end calculateDifficultyRating
