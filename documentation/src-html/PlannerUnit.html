135    // provides a sorting value, with a smaller number indicating an

137    public int sortValue() {
138        int value;
139        value = year*4;
140        
141        if(semester.equals("SPRING"))
142            value = value+0;
143        else if (semester.equals("SUMMER"))
144            value = value+1;
145        else if (semester.equals("FALL"))
146            value = value+2;
147        else
148            value = value+3;
149
